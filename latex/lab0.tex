\begin{frame}
\frametitle{Personal Learning Goals}
    \textbf{On The Whiteboard:}
    \begin{itemize}
        \item \textbf{\textcolor{blue}{Brainstorm} personal goals related to CSC 212}
        \begin{itemize}
            \item \textbf{Ex:} I will get a final grade of 85\% or higher in CSC 212.
            \item \textbf{Ex:} I will pass a technical interview for an internship. 
            \item \textbf{Ex:} I will attend office hours at least once a week.
        \end{itemize}
    \end{itemize}
    
    \textbf{Ensure Your Goals Are:}
    \begin{itemize}
        \item \textbf{\textcolor{red}{Specific.}} What will be accomplished? What actions will you take?
        \item \textbf{\textcolor{red}{Measurable.}} What data will measure the goal? How do you know you have succeeded?
        \item \textbf{\textcolor{red}{Achievable.}} Is the goal doable? Do you have the necessary skills and resources?
        \item \textbf{\textcolor{red}{Relevant.}} How does the goal align with broader goals? Why is the result important?
        \item \textbf{\textcolor{red}{Time-Bound.}} Can you accomplish your goal this semester?
    \end{itemize}
\end{frame}
